import time
def time_lapse(func):
    def wrapper(*args, **kwargs):
        start = time.time()  
        result = func(*args, **kwargs)

        end = time.time()    

        print("걸린시간 %d" % (end - start))
        return result
    
    return wrapper        

factorial(10)

@time_lapse
def factorial(num):
    if num <= 1:
        return 1
    return num * factorial(num - 1)


fact = time_lapse(factorial)

fact(10)
